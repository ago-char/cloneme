<!-- get current working dir  -->
os.getcwd()

<!-- list files/folders  -->
os.listdir()

<!-- change dir -->
os.chdir(<path>)

<!-- create dir  -->
os.mkdir(dir/path)

<!-- remove dir  -->
os.rmdir(dir/path)

<!-- create tree -->
os.makedirs(tree/path)

<!-- remove tree -->
os.removedirs(tree/path)

<!-- rename file -->
os.rename(originalName,newName)

<!-- create file  -->
os.mknod('filename')

<!-- remove file  -->
os.remove('filename')
